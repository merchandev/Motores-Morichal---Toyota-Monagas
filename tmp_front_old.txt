  <script>
    (function() {
      window.addEventListener('load', function () {
        var slider = document.getElementById('custom-slider');
        if (!slider) return;

        var videos = slider.querySelectorAll('video');
        var progressSpans = slider.querySelectorAll('.cs-progress-bar span');
        var lastVideoListener = null;

        var swiper = new Swiper('#custom-slider', {
          loop: false,
          navigation: { nextEl: '.cs-swiper-button-next', prevEl: '.cs-swiper-button-prev' },
          allowTouchMove: true,
          observer: true,
          observeParents: true,
          on: {
            init: function () { updateProgress(this.realIndex); },
            slideChange: function () { updateProgress(this.realIndex); }
          }
        });

        function resetProgress() {
          progressSpans.forEach(function(span) { span.style.width = '0%'; });
          if (lastVideoListener && lastVideoListener.video) {
            lastVideoListener.video.removeEventListener('timeupdate', lastVideoListener.handler);
            lastVideoListener.video.removeEventListener('ended', lastVideoListener.endedHandler);
            lastVideoListener = null;
          }
        }

        function updateProgress(index) {
          resetProgress();
          var vid = videos[index];
          var bar = progressSpans[index];
          if (!vid || !bar) return;
          vid.currentTime = 0;
          vid.play().catch(function(){});
          var handler = function() {
            if (!vid.duration) return;
            var percent = (vid.currentTime / vid.duration) * 100;
            bar.style.width = percent + '%';
          };
          var endedHandler = function() {
            bar.style.width = '100%';
          };
          vid.addEventListener('timeupdate', handler);
          vid.addEventListener('ended', endedHandler);
          lastVideoListener = { video: vid, handler: handler, endedHandler: endedHandler };
        }

        updateProgress(swiper.realIndex);
      });
    })();
  </script>

  <!-- Swiper JS -->
  <script src="https://cdn.jsdelivr.net/npm/swiper@11/swiper-bundle.min.js"></script>
  <script>
    (function(){
      const wrapper = document.querySelector('#vehiculos .toyota-slider .swiper-wrapper');
      const templates = Array.from(document.querySelectorAll('#vehiculos .toyota-templates template'));
      const tabs = document.querySelectorAll('#vehiculos .toyota-tab');
      const indicator = document.querySelector('#vehiculos .toyota-tab-indicator');

      const swiper = new Swiper('#vehiculos .toyota-slider', {
        spaceBetween: 20,
        navigation: {
          nextEl: '#vehiculos .toyota-arrow.swiper-button-next',
          prevEl: '#vehiculos .toyota-arrow.swiper-button-prev'
        },
        breakpoints: { 0:{slidesPerView:1}, 768:{slidesPerView:2}, 1024:{slidesPerView:3} }
      });

      function buildSlides(category='all'){
        wrapper.innerHTML = '';
        const items = templates.filter(t => category==='all' || (t.dataset.cat||'').includes(category));
        items.forEach(tpl=>{
          const slide = document.createElement('div');
          slide.className='swiper-slide';
          slide.appendChild(tpl.content.cloneNode(true));
          wrapper.appendChild(slide);
        });
        swiper.update();
      }

      function moveIndicator(tab){
        const rect = tab.getBoundingClientRect();
        const parentRect = tab.parentElement.getBoundingClientRect();
        indicator.style.width = rect.width+'px';
        indicator.style.transform = 'translateX(' + (rect.left - parentRect.left) + 'px)';
      }

      tabs.forEach(tab=>{
        tab.addEventListener('click', ()=>{
          document.querySelector('#vehiculos .toyota-tab.is-active')?.classList.remove('is-active');
          tab.classList.add('is-active');
          buildSlides(tab.dataset.cat || 'all');
          moveIndicator(tab);
        });
      });

      // Inicializar indicador
      const activeTab = document.querySelector('#vehiculos .toyota-tab.is-active');
      if(activeTab) moveIndicator(activeTab);
      // Construir los slides iniciales
      buildSlides();
    })();
  </script>
